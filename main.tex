\begin{equation*}
    \Psi(x,y) = \frac{2}{a} \sin \left( \frac{n\pi x}{a} \right) \sin \left( \frac{m \pi y}{a} \right)
\end{equation*}

    \text{with the corresponding energies being given by: }
    
\begin{equation*}
    E_{nm} = (n^2 + m^2) \frac{\pi^2 h^2}{2m a^2}
